.data
	prompt: .asciiz "Enter the string."
	message: .asciiz "\nThe converted string is "
.text
main:
	li $v0, 4
	la $a0, prompt
	syscall
	#prompt user for string
	li $v0,5
	syscall
	move $t0, $v0
	jal strlen
	addi $v1, $v0, 0

	li $v0, 4
	la $a0, message
	syscall
	li $v0, 1
	move $a0, $t0
